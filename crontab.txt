# 定时任务配置
# 格式：分钟 小时 日期 月份 星期 命令

# 每天凌晨2点执行
0 2 * * * cd /app/src && python main.py >> /app/data/cron.log 2>&1

# 每小时执行一次
0 * * * * cd /app/src && python main.py >> /app/data/cron.log 2>&1

# 每30分钟执行一次
*/30 * * * * cd /app/src && python main.py >> /app/data/cron.log 2>&1

# 每周一早上9点执行
0 9 * * 1 cd /app/src && python main.py >> /app/data/cron.log 2>&1

# 每月1号凌晨1点执行
0 1 1 * * cd /app/src && python main.py >> /app/data/cron.log 2>&1
